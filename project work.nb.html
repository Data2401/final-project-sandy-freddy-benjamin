gaming_data <- gaming_data %>%
  mutate(
    suspicious = case_when(
      daily_gaming_hours > quantile(daily_gaming_hours, 0.75, na.rm = TRUE) &
      addiction_level < quantile(addiction_level, 0.25, na.rm = TRUE) ~ "Underreporting",

      daily_gaming_hours < quantile(daily_gaming_hours, 0.25, na.rm = TRUE) &
      addiction_level > quantile(addiction_level, 0.75, na.rm = TRUE) ~ "Overreporting",

      TRUE ~ "Consistent"
    )
  )
gaming_data %>%
  count(suspicious)
